} } function decreaseQuantity(event) { const cartItem = event.target.closest(".cart-item"); const productId = cartItem .querySelector(".quantity-button.minus") .getAttribute("data-product"); const product = getProductDetails(productId); return productsData.find((product) => product.id === productId); } function increaseCartItemQuantity(product) { const existingProduct = cart.find((item) => item.id === product.id); if (existingProduct) { existingProduct.quantity++; } } function decreaseCartItemQuantity(product) { const existingProduct = cart.find((item) => item.id === product.id); if (existingProduct && existingProduct.quantity > 1) { existingProduct.quantity--; } } function updateCartDisplay() { // Cập nhật giao diện hiển thị giỏ hàng tại đây. // Bạn có thể sử dụng DOM manipulation để thay đổi nội dung giỏ hàng. } }); // Lấy danh sách nút cộng và trừ const plusButtons = document.querySelectorAll(".quantity-button.plus"); const minusButtons = document.querySelectorAll(".quantity-button.minus"); plusButtons.forEach(button => { button.addEventListener("click", increaseQuantity); }); minusButtons.forEach(button => { button.addEventListener("click", decreaseQuantity); }); function increaseQuantity(event) { const cartItem = event.target.closest(".cart-item"); const quantityText = cartItem.querySelector(".quantity-text"); let currentQuantity = parseInt(quantityText.textContent); currentQuantity++; quantityText.textContent = currentQuantity; } function decreaseQuantity(event) { const cartItem = event.target.closest(".cart-item"); const quantityText = cartItem.querySelector(".quantity-text"); let currentQuantity = parseInt(quantityText.textContent); if (currentQuantity > 1) { currentQuantity--; quantityText.textContent = currentQuantity; } } // Lấy danh sách nút Xóa const removeButtons = document.querySelectorAll(".remove-button"); // Xóa sản phẩm khỏi giỏ hàng dựa trên productId removeFromCart(productId); // Xóa sản phẩm khỏi giao diện cartItem.remove(); // Cập nhật hiển thị tổng tiền hoặc các thông tin khác của giỏ hàng updateCartDisplay(); } function removeFromCart(productId) { // Xóa sản phẩm khỏi mảng giỏ hàng dựa trên productId const productIndex = cart.findIndex(item => item.id === productId); if (productIndex !== -1) { cart.splice(productIndex, 1);
